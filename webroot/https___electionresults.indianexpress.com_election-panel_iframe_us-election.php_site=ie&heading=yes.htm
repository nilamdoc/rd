   function animationD(){
   const ready = (selector, callback) => {
    [...document.querySelectorAll(selector)].forEach(callback);
};

ready('.counter', (stat) => {
   const patt = /(\D+)?(\d+)(\D+)?(\d+)?(\D+)?/;
   const time = 200;
   let result = [...patt.exec(stat.textContent)];
   let fresh = true;
   let ticks;
   result.shift();
   while (stat.firstChild) {
      stat.removeChild(stat.firstChild);
   }

   for (let res of result) {
      if (isNaN(res)) {
      }
      else {
         for (let i = 0; i < res.length; i++) {
            stat.insertAdjacentHTML('beforeend',
               `<div class="animation" data-value="${res[i]}">
                  <b>0</b>
                  ${Array(parseInt(res[i]) + 1).join(0).split(0).map((x, j) => `
                     <b>${j}</b>
                  `).join('')}
               </div>`
            );
         }
      }
   }

   ticks = [...stat.querySelectorAll('div[data-value]')];

   let activate = () => {
      let top = stat.getBoundingClientRect().top;
      let offset = (window.innerHeight * 3 / 4);
      if (top < offset) {
            for (let tick of ticks) {
               let dist = parseInt(tick.getAttribute('data-value')) + 1;
               tick.style.transform = `translateY(-${(dist) * 100}%)`
            }
      }
   }
   activate();
});
}

<script>
function UsElection( $site, $headers ) {
   var staticUrl = 'https://electionresults.indianexpress.com/json/us-elections-2024-chart.json';
   fetch(staticUrl)
      .then(response => response.json())
      .then(data => {
         const states = ['uselection'];
         let us_election_widget = '';

         states.forEach(state => {
            const assembly = data[state].story;
            const items = data[state].uselection.items;
            if ( 'yes' === $headers ) {
               us_election_widget += `
                  <div class="us_heading">
                     <h1 class="heading"><a href="${assembly.storyURL}" target="_blank">${assembly.storyTitle}</a></h1>
                     <p>${assembly.storyDescription}</p>
                  </div>`
            }
            else if ( 'no' === $headers ) {
               us_election_widget += `
                  <div class="us_heading">
                     <h1 class="heading">US Election Result 2024</h1>
                  </div>`
            }
            us_election_widget +=
               `<section class="leader_container">
                  <div class="leader trump">
                        <img src="https://data.indianexpress.com/election2019/election2019/assemblyelection/images/trump.png" />
                  </div>
                  <div class="result_box">
                        <span class="democrats_party party">
                           <em class="counter">${items[0].pTotalseats}</em> ${items[0].pName}
                        </span>
                        <div class="result_progress_bar">
                           <span class="republicans" style="--width:${items[0].totalpercent}%"></span>
                           <span class="democrats" style="--width:${items[1].totalpercent}%"></span>
                        </div>
                        <span class="republicans_party party">
                           <em class="counter">${items[1].pTotalseats}</em> ${items[1].pName}
                        </span>
                        <span class="total_value">${items[0].percentage} <em></em></span>
                        <span class="total_value">${items[1].percentage} <em></em></span>
                  </div>
                  <div class="leader harris">
                        <img src="https://data.indianexpress.com/election2019/election2019/assemblyelection/images/harris.png" />
                  </div>
               </section>`;
         });

         // Update the HTML with the generated content
         document.querySelector('.us_result').innerHTML = us_election_widget;
      //animationD();
      })
      .catch(error => console.error('Error fetching data:', error));
}

document.addEventListener('DOMContentLoaded', function() {
   var $site    = "ie";
   var $headers = "yes";
   UsElection($site, $headers);
   setInterval(function() {
      UsElection($site, $headers);
   }, 60000); // Refresh every 60 seconds
});
</script>
